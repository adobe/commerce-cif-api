set $ow_namespace "your-namespace";

location = /commerce/carts {
  if ($request_method != POST) {
    return 405;
  }
  rewrite ^ /api/v1/web/$ow_namespace/default/postCart.http break;
  proxy_pass $controller;
  proxy_read_timeout 70s;
}
